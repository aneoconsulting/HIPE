184  inline void emitPacket(scy::av::VideoDecoder* dec, AVFrame* frame)
185  {
186  frame->pts = av_frame_get_best_effort_timestamp(frame);
187 
188  // Set the decoder time in microseconds
189  // This value represents the number of microseconds
190  // that have elapsed since the brginning of the stream.
191 
192  dec->time = frame->pts > 0 ? static_cast<int64_t>(frame->pkt_pts *
193  av_q2d(dec->stream->time_base) * AV_TIME_BASE) : 0;
194 
195  // Set the decoder pts in stream time base
196  dec->pts = frame->pts;
